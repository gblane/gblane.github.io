About

Analytical solutions for diffuse reflectance from a semi-infinite medium using the extrapolated boundary condition. Switch between Continuous-Wave (CW), Frequency-Domain (FD), and Time-Domain (TD).

Background

Semi-infinite diffuse reflectance with extrapolated boundary condition

Consider a pencil beam incident on the surface of a semi-infinite, homogeneous, turbid medium. Within the diffusion approximation the pencil beam is replaced by an isotropic point source placed one transport mean free path below the surface at depth \(z_0 = 1/\mu_{tr}\), where \(\mu_{tr} = \mu_a + \mu_s'\). The diffusion coefficient is \(D = 1/(3\mu_{tr})\). To satisfy the extrapolated boundary condition an image (virtual) source mirrors the real source across the extrapolated boundary at \(z_b = -2AD\), giving image height \(z_0' = -z_0 + 2z_b\). The parameter \(A\) accounts for the refractive-index mismatch between the medium (\(n_{in} = 1.4\)) and air (\(n_{out} = 1\)), giving \(A \approx 2.95\). With the detector on the surface at source–detector separation \(\rho\), the distances to the real and image sources are \(r_1 = \sqrt{\rho^2 + z_0^2}\) and \(r_2 = \sqrt{\rho^2 + z_0'^{\,2}}\).

Continuous-Wave (CW)

The effective attenuation is \(\mu_{eff} = \sqrt{\mu_a/D}\) and the CW diffuse reflectance is:

$$R_{CW}(\rho) = \frac{1}{4\pi} \left[ z_0 \frac{\mu_{eff} + 1/r_1}{r_1^2}\,e^{-\mu_{eff} r_1} - z_0' \frac{\mu_{eff} + 1/r_2}{r_2^2}\,e^{-\mu_{eff} r_2} \right]$$

Frequency-Domain (FD)

For an intensity-modulated source at angular frequency \(\omega = 2\pi f_{mod}\), the effective attenuation becomes complex,

$$\tilde\mu_{eff} = \sqrt{\frac{\mu_a}{D} - \frac{i\,\omega}{v\,D}}, \qquad v = c/n_{in}$$

and the complex reflectance \(\tilde R_{FD}(\rho)\) takes the same form as \(R_{CW}\) with \(\mu_{eff}\) replaced by \(\tilde\mu_{eff}\). Magnitude \(|\tilde R_{FD}|\) and phase \(\angle \tilde R_{FD}\) are the typical observables.

Time-Domain (TD)

For an impulse source the diffuse reflectance versus time \(t > 0\) is:

$$R_{TD}(\rho,t) = \frac{1}{2}\, \frac{e^{-\mu_a v t}}{(4\pi D v)^{3/2}\, t^{5/2}} \left[ z_0\,e^{-r_1^2/(4Dvt)} + (z_0 - 2 z_b)\,e^{-r_2^2/(4Dvt)} \right]$$

Units: CW and FD reflectance are mm⁻²; TD reflectance is (ps mm²)⁻¹. All expressions are valid in the diffusion regime, i.e., \(\rho \gg \ell_{tr} = 1/\mu_{tr}\) and \(\mu_a \ll \mu_s'\).

// Refractive-index mismatch parameter A(n)
function n2A(n) {
    if (n > 1) {
        return 504.332889 - 2641.00214*n + 5923.699064*n**2
             - 7376.355814*n**3 + 5507.53041*n**4
             - 2463.357945*n**5 + 610.956547*n**6
             - 64.8047*n**7;
    } else if (n < 1) {
        return 3.084635 - 6.531194*n + 8.357854*n**2
             - 5.082751*n**3 + 1.171382*n**4;
    }
    return 1;
}

const N_IN  = 1.4;
const A     = n2A(N_IN);
const V_PS  = 0.299792458 / N_IN;     // mm/ps    (for TD: t in ps)
const V_SEC = 2.99792458e11 / N_IN;   // mm/sec   (for FD: ω in rad/sec)

// ── Geometry helpers ─────────────────────────────────
function geom(mua, musp) {
    const mutr = mua + musp;
    const D    = 1 / (3 * mutr);
    const z0   = 1 / mutr;
    const zb   = -2 * A * D;
    const zp   = -z0 + 2 * zb;  // image-source height
    return { D, z0, zb, zp };
}

// ── CW: real-valued reflectance ──────────────────────
function R_CW(rho, mua, musp) {
    const { D, z0, zp } = geom(mua, musp);
    const mueff = Math.sqrt(mua / D);
    const r1 = Math.hypot(rho, z0);
    const r2 = Math.hypot(rho, zp);
    const t1 =  z0 * (mueff + 1/r1) * Math.exp(-mueff*r1) / (r1*r1);
    const t2 = -zp * (mueff + 1/r2) * Math.exp(-mueff*r2) / (r2*r2);
    return (t1 + t2) / (4 * Math.PI);
}

// ── FD: complex reflectance (returns {re, im}) ───────
function csqrt(re, im) {
    const r = Math.hypot(re, im);
    const a = Math.sqrt((r + re) / 2);
    const b = Math.sign(im || 1) * Math.sqrt((r - re) / 2);
    return { re: a, im: b };
}
function R_FD(rho, mua, musp, omega) {
    const { D, z0, zp } = geom(mua, musp);
    // mueff = sqrt(mua/D - i*omega/(v*D)) with v in mm/sec, omega in rad/sec
    const mu = csqrt(mua / D, -omega / (V_SEC * D));
    const r1 = Math.hypot(rho, z0);
    const r2 = Math.hypot(rho, zp);

    // term(r, zsign): zsign * (1/r + mu) * exp(-mu*r) / r^2
    function term(r, zsign) {
        const mr = -mu.re * r, mi = -mu.im * r;       // -mu*r
        const e  = Math.exp(mr);
        const ex = { re: e * Math.cos(mi), im: e * Math.sin(mi) };
        const fac = { re: 1/r + mu.re, im: mu.im };
        const num = { re: fac.re*ex.re - fac.im*ex.im,
                      im: fac.re*ex.im + fac.im*ex.re };
        return { re: zsign * num.re / (r*r), im: zsign * num.im / (r*r) };
    }
    const a = term(r1,  z0);
    const b = term(r2, -zp);
    return { re: (a.re + b.re) / (4*Math.PI),
             im: (a.im + b.im) / (4*Math.PI) };
}

// ── TD: impulse reflectance ──────────────────────────
function R_TD(rho, t, mua, musp) {           // t in ps
    const { D, z0, zb, zp } = geom(mua, musp);
    const v   = V_PS;
    const pre = Math.exp(-mua * v * t)
              / (Math.pow(4*Math.PI*D*v, 1.5) * Math.pow(t, 2.5));
    const r1sq = rho*rho + z0*z0;
    const r2sq = rho*rho + zp*zp;
    const s = z0 * Math.exp(-r1sq / (4*D*v*t))
            + (z0 - 2*zb) * Math.exp(-r2sq / (4*D*v*t));
    return 0.5 * pre * s;
}
